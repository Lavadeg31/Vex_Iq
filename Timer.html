      <div class="plus-minus">

      </div>

        <button onclick="adjustValue('goals', 1)">+</button>

        <input type="number" id="switches-display" value="0" min="0" max="4" oninput="calculatePoints()">

        <button onclick="adjustValue('switches', 1)">+</button>

      </div>

  <script>

    let mode = 'teamwork'; // default mode


        function toggleTimer() {

            const timerSection = document.getElementById("timer-section");

            timerSection.classList.toggle("hidden");

        }


        function startTimer() {

            let countdown = 3;

            let timerDisplay = document.getElementById("timer");

            let audio = document.getElementById("timerAudio");


            // Play the audio immediately when the button is pressed

            audio.play();


            // Countdown before the timer starts

            let countdownInterval = setInterval(() => {

                countdown--;

                timerDisplay.innerHTML = countdown;

                if (countdown === 0) {

                    clearInterval(countdownInterval);

                    startMinuteTimer(); // Start the 1-minute timer after countdown

                }

            }, 1000);

        }


        function startMinuteTimer() {

            let timeLeft = 60;

            let timerDisplay = document.getElementById("timer");


            let timerInterval = setInterval(() => {

                timeLeft--;

                timerDisplay.innerHTML = timeLeft;

                if (timeLeft === 0) {

                    clearInterval(timerInterval);

                    timerDisplay.innerHTML = "Time's up!";

                }

            }, 1000);

        }


        function selectMode(selectedMode) {

            mode = selectedMode;


            document.getElementById('teamwork-button').classList.remove('active');

            document.getElementById('skills-button').classList.remove('active');

            document.getElementById(selectedMode + '-button').classList.add('active');


            if (mode === 'teamwork') {

                document.getElementById('passes-group').classList.remove('hidden');

            } else if (mode === 'skills') {

                document.getElementById('passes-group').classList.add('hidden');

            }


            calculatePoints();

        }


        function adjustValue(id, adjustment) {

            let currentValue = parseInt(document.getElementById(id + '-display').value);

            let newValue = Math.max(0, currentValue + adjustment);

            if (id === 'switches') newValue = Math.min(4, newValue);

            document.getElementById(id + '-display').value = newValue;

            calculatePoints();

        }


        function calculatePoints() {

            let passes = parseInt(document.getElementById('passes-display').value) || 0;

            let goals = parseInt(document.getElementById('goals-display').value) || 0;

            let switches = Math.min(parseInt(document.getElementById('switches-display').value) || 0, 4);

            

            let totalPoints = 0;


            if (mode === 'teamwork') {

                totalPoints = (passes * 2) + (goals * 5) + (switches * 3);

            } else if (mode === 'skills') {

                totalPoints = switches * 1; // 1 point per switch

                switch (switches) {

                    case 1:

                        totalPoints += goals * 4;

                        break;

                    case 2:

                        totalPoints += goals * 8;

                        break;

                    case 3:

                        totalPoints += goals * 10;

                        break;

                    case 4:

                        totalPoints += goals * 12;

                        break;

                }

            }


            document.getElementById('result').textContent = 'Total Points: ' + totalPoints;

        }


        function resetValues() {

            document.getElementById('passes-display').value = 0;

            document.getElementById('goals-display').value = 0;

            document.getElementById('switches-display').value = 0;

            calculatePoints();

            }

            

            calculatePoints(); // Initialize points
